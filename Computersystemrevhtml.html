<script>

    // --- DATABASE ---

    const questions = [

        { q: "Which register holds the address of the next instruction?", a: "Program Counter", options: ["MAR", "MDR", "Program Counter", "Accumulator"], diff: "easy" },

        { q: "Which memory is non-volatile?", a: "ROM", options: ["RAM", "ROM", "Cache", "Register"], diff: "easy" },

        { q: "A 3D printer is an example of what?", a: "Output Device", options: ["Input Device", "Output Device", "Storage", "Processing"], diff: "easy" },

        { q: "What is the primary purpose of defragmentation?", a: "Rearrange files to be contiguous", options: ["Delete viruses", "Compress files", "Rearrange files to be contiguous", "Increase clock speed"], diff: "hard" },

        { q: "Which bus carries signals to coordinate the fetch-execute cycle?", a: "Control Bus", options: ["Data Bus", "Address Bus", "Control Bus", "USB"], diff: "hard" }

    ];

 

    const longQuestions = [

        "Explain the difference between volatile and non-volatile memory with examples. (4 marks)",

        "Describe the IPO cycle and how it relates to a barcode scanner system. (6 marks)",

        "Compare the pros and cons of Cloud Storage versus Local Secondary Storage. (6 marks)"

    ];

 

    let currentQuizSet = [];

    let currentIdx = 0;

    let score = 0;

 

    // --- NAVIGATION ---

    function showSection(id) {

        document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));

        document.getElementById(id).classList.remove('hidden');

    }

 

    // --- QUIZ LOGIC ---

    function startQuiz() {

        const diff = document.getElementById('mc-diff').value;

        const qty = document.getElementById('mc-qty').value;

       

        currentQuizSet = questions

            .filter(q => q.diff === diff)

            .sort(() => 0.5 - Math.random())

            .slice(0, qty);

       

        currentIdx = 0;

        score = 0;

        showSection('quiz-game');

        renderQuestion();

    }

 

    function renderQuestion() {

        const q = currentQuizSet[currentIdx];

        document.getElementById('q-text').innerText = `Question ${currentIdx + 1}: ${q.q}`;

        const optContainer = document.getElementById('options-container');

        optContainer.innerHTML = '';

        document.getElementById('feedback').classList.add('hidden');

 

        q.options.sort(() => 0.5 - Math.random()).forEach(opt => {

            const btn = document.createElement('button');

            btn.className = 'option-btn';

            btn.innerText = opt;

            btn.onclick = () => checkAnswer(opt, btn);

            optContainer.appendChild(btn);

        });

    }

 

    function checkAnswer(choice, btn) {

        const correct = currentQuizSet[currentIdx].a;

        const btns = document.querySelectorAll('.option-btn');

        btns.forEach(b => b.disabled = true);

 

        if (choice === correct) {

            btn.classList.add('correct');

            score++;

            document.getElementById('feedback-msg').innerHTML = "Correct!";

        } else {

            btn.classList.add('wrong');

            document.getElementById('feedback-msg').innerHTML = `Incorrect. The correct answer was: <strong>${correct}</strong>.`;

        }

        document.getElementById('feedback').classList.remove('hidden');

    }

 

    function nextQuestion() {

        currentIdx++;

        if (currentIdx < currentQuizSet.length) {

            renderQuestion();

        } else {

            const grade = (score / currentQuizSet.length) * 100;

            let dseGrade = grade >= 80 ? "5**" : grade >= 70 ? "5" : grade >= 50 ? "4" : "U";

            document.getElementById('question-container').innerHTML = `<h2>Quiz Complete!</h2><p>Score: ${score}/${currentQuizSet.length}</p><p>Predicted Grade: ${dseGrade}</p>`;

            document.getElementById('feedback').classList.add('hidden');

        }

    }

 

    // --- MOCK PAPER LOGIC ---

    function generateMock() {

        showSection('mock-paper');

        const content = document.getElementById('paper-content');

        const diff = document.getElementById('mock-diff').value;

       

        let html = `<h1 style="text-align:center">ICT MOCK EXAMINATION</h1>`;

        html += `<p style="text-align:center">Difficulty: ${diff.toUpperCase()}</p><hr>`;

       

        html += `<h3>SECTION A: Multiple Choice (5 Marks)</h3>`;

        questions.forEach((q, i) => {

            html += `<p>${i+1}. ${q.q}<br>`;

            q.options.forEach(o => html += `&nbsp;&nbsp;&nbsp;[ ] ${o}<br>`);

            html += `</p>`;

        });

 

        html += `<hr><h3>SECTION B: Conventional Questions (16 Marks)</h3>`;

        longQuestions.forEach((q, i) => {

            html += `<p><strong>Question ${i+1}</strong><br>${q}</p><br><br><br>`;

        });

 

        content.innerHTML = html;

    }

</script>
